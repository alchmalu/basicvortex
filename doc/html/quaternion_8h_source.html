37 template <typename T>
38 glm::detail::tquat<T> from_to(
39  glm::detail::tvec3<T> const & from,
40  glm::detail::tvec3<T> const & to){
41  const double epsilon = 1E-10f;
42  const double fromSqNorm = glm::dot(from, from);
43  const double toSqNorm = glm::dot(to, to);
44  // Identity Quaternion when one vector is null
45  if ((fromSqNorm < epsilon) || (toSqNorm < epsilon)) {
46  return glm::detail::tquat<T>(0,0,0,1);
47  } else {
48  glm::vec3 axis = glm::cross(from, to);
49  const double axisSqNorm = glm::dot(axis, axis);
50 
51  // Aligned vectors, pick any axis, not aligned with from or to
52  if (axisSqNorm < epsilon)
53  axis = orthogonalVector(from);
54 
55  double angle = asin(sqrt(axisSqNorm / (fromSqNorm * toSqNorm)));
56 
57  if (glm::dot(from,to) < 0.0)
58  angle = M_PI-angle;
59 
60  glm::detail::tquat<T>(angle, axis);
61  }
62 }

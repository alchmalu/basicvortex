113 inline glm::vec3 perpendicular (const glm::vec3 & v) {
114  static const float fSquareZero = 1e-06 * 1e-06;
115  glm::vec3 perp = glm::cross(v, glm::vec3(1,0,0));
116 
117  if (sqlength(perp)< fSquareZero) {
118  /* This vector is the Y axis multiplied by a scalar, so we have
119  to use another axis.
120  */
121  perp = glm::cross(v, glm::vec3(0,1,0));
122  }
123 
124  return perp;
